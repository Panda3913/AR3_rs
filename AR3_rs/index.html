Controlling the AR3 robot arm.

Examples

let ar3_info: String = AR3_rs::data_pub(&mut sp, "?".to_string()).unwrap();
println!("{:?}",ar3_info);

python

from AR3_rs import Ar3,delta_theta,moving_time,velocity,check_ar3_port,ar3_port,log_init,log_print,set_level

LOG = log_print

A = [156,36,55,86,72,154]       # virtual zeros
B = [156,36,55,86,160,154]      # standard posture

## Log ###
log_init()
LOG('info',"info msg")

## port ###
print(ar3_port())               # COM3
print(check_ar3_port("COM3"))   # True

## Calculate arm information ###
dth = delta_theta(A,B)
print(dth)                      # 88.0
print(moving_time(dth,5000))    # 1.056
print(velocity(dth,1.056))      # 5000.0

## Control arms and get information ###
arm=Ar3()
print(arm.ar3_port)             # COM3
print(arm.data_pub("?"))        # <Idle|MPos:156.001,36.000,54.999,86.000,159.997,154.000,0.000,0.000|Bf:15,510|FS:0,0|Pn:S>
print(arm.goto0())              # ok
print(arm.zero_all())           # Found no data!
print(arm.get_all())            # [156.001, 36.0, 54.999, 86.0, 159.997, 154.0]
print(arm.servo(50))            # ok
print(arm.get_servo())          # 50
print(arm.set_arm(90,A,5000))   # ok
